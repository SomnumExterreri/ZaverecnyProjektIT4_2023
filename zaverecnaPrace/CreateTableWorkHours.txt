CREATE TABLE [dbo].[WorkHours] (
    [Id]            INT          NOT NULL,
    [PersonalNumber]INT          NOT NULL,
    [ContactId]     INT          NOT NULL,
    [WorkTypeStyleId] INT          NOT NULL,
    [Hours]         INT          NOT NULL,
    PRIMARY KEY CLUSTERED ([Id] ASC),
    CONSTRAINT [EmployeeId] FOREIGN KEY ([PersonalNumber]) REFERENCES [dbo].[Employee] ([PersonalNumber]),
    CONSTRAINT [ContactId] FOREIGN KEY ([ContactId]) REFERENCES [dbo].[Contract] ([ContractNumber]), 
    CONSTRAINT [WorkTypeStyle] FOREIGN KEY ([WorkTypeStyleId]) REFERENCES [dbo].[WorkType]([Id])
);
